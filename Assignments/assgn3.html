Word Guesser 

# ENTER WORD TO BE GUESSED HERE
WORD_PERMANENT = "Internet"

# storing mutable and lowercased version of word
word = WORD_PERMANENT.lower()

# declare set for guessed letters
guessedLetters = set()

# run until break
while True:
    # collect user input
    letter = input('Guess a letter: ')
    # try, assuming a single letter was input
    try:
        assert len(letter) == 1
        # make letter lowercase
        letter = letter.lower()
        # if word contains the letter
        if letter in word:
            print("Correct. ", letter, " is in the word")
            # add letter to guessed set
            guessedLetters.add(letter)
            # replace letter in mutable version of word with a space
            word = word.replace(letter, " ")
        else:
            print("Nope. Word either doesn't contain that letter, or you have",
                  "already guessed it.")
        # prints remaining ammount of letters to guess
        print("left to guess: ", word.count(' '))
        # prints word status, for example: "Int_rn_t"
        print("guesses so far: ")
        for index, char in enumerate(word):
            if char == ' ':
                print(WORD_PERMANENT[index], end='')
            else:
                print('_', end='')
        print("\n")
        # break case where all letters are guessed
        if word.count(' ') == 0:
            print("Done. You have guessed the word: ", WORD_PERMANENT)
            break
    except AssertionError:
        print("Input string must be of size 1\n\n")


Weather API Request 

import json
import requests
from datetime import datetime as dt

# pep8 compliant ;)

ZIP = 30060
COUNTRY = 'us'

# gathering key from outside text file
with open('key.txt', 'r') as file:
    KEY = file.read()

# creating url string for api request
apiString = 'https://api.openweathermap.org/data/2.5/weather?'
apiString += 'zip={},{}&appid={}'.format(ZIP, COUNTRY, KEY)

# storing API results in 'data'
data = requests.get(apiString).json()

# using datetime to gather time of report from unix timestamp
dtObj = dt.fromtimestamp(data['dt'])
dtStr = dtObj.strftime('%Y-%m-%d %H:%M:%S')

# creating a dictionary of all the necessary parts of the data
myDict = {'Name': data['name'],
          'Current Temperature':
          str(round(1.8 * (float(data['main']['temp']) - 273) + 32, 2)) +
          ' degrees Fahrenheit',
          'Atmospheric Pressure': str(round(data['main']['pressure'], 2)) +
          ' hPa',
          'Wind Speed': str(data['wind']['speed']) + ' mph',
          'Wind Direction': data['wind']['deg'],
          'Time of Report': dtStr
          }

# print each piece of the dictionary
for k in myDict:
    print(k.ljust(25) + str(myDict[k]))
