import sys
    import requests
    
    # initialize vars
    webUrl, outputPath = str(), str()
    
    # try to store GET response
    try:
        webUrl, outputPath = tuple(sys.argv[1:3])
        response = requests.get(webUrl)
    except requests.exceptions.MissingSchema as missingSchema:
        sys.exit(missingSchema)
    except requests.exceptions.ConnectionError as connError:
        sys.exit(connError)
    except ValueError as e:
        print('Arguments expected: 2\nArguments received:', str(len(sys.argv) - 1))
        print('Argument pattern: client.py \"WEB_URL>\" \"<OUTPUT_FILE>\"')
    
    # try to see result as JSON and if not as text
    try:
        result = response.json()
    except ValueError:
        result = response.text
    
    # write to output file
    with open(outputPath, 'w') as output:
        output.write(result)
